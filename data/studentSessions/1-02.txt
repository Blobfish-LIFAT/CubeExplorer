SELECT NON EMPTY {Hierarchize({{[Measures].[Distance trajet domicile - travail (max)], [Measures].[Distance trajet domicile - travail (min)], [Measures].[Distance trajet domicile - travail (moyenne)]}})} ON COLUMNS, NON EMPTY CrossJoin({[Commune de residence.CNERES_Hierarchie_intercommunale].[CENTRE].[Pays des Châteaux].[AGGLOPOLYS]}, CrossJoin([Statut d'emploi.STATEMPL_Hierarchie].[Statut d'emploi de niveau 1].Members, [Sexe.Sexe_Hierarchie].[Sexe].Members)) ON ROWS FROM [Cube1MobProInd] 

--On remarque que les hommes font plus de trajet que les femmes, et cela s'explique car les femmes préfèrent être plus proches du domicile pour s'occuper de leurs enfants et la maison

SELECT NON EMPTY {Hierarchize({{[Measures].[Distance trajet domicile - travail (max)], [Measures].[Distance trajet domicile - travail (min)], [Measures].[Distance trajet domicile - travail (moyenne)]}})} ON COLUMNS, NON EMPTY CrossJoin({[Commune de residence.CNERES_Hierarchie_intercommunale].[CENTRE].[Pays des Châteaux].[AGGLOPOLYS]}, CrossJoin([Statut d'emploi.STATEMPL_Hierarchie].[Statut d'emploi de niveau 1].Members, CrossJoin([Temps d'emploi.TEMPSEMP_Hierarchie].[Temps d'emploi].Members, [Mode de transport.MODTRANS_Hierarchie].[Mode de transport].Members))) ON ROWS FROM [Cube1MobProInd] 

--On remarque que le trajet effectuer en transport en commun est le plus long, et c'est la même chose pour d'autres communes qu'aggropolys. Les transports en communs coutent moins cher que la voiture, motos... pour le long trajet c'est plus intéressant financièrement de prendre les transports en communs. Une autre raison possible c'est que les transports en communs ne font pas un trajet direct comme on pourrai le faire avec une voiture. Les transports en commun font des détours et cela augmente donc la distance parcouru pour aller au travail.
